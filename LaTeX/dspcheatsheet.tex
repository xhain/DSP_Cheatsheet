		\[

			1,& \text{if }  n = 0\\
			\text{0 }  & \text{otherwise}
		\end{cases}
		\]
		
		unit step:
		\[
		u(n)= 
		\begin{cases}
		1,& \text{if }  n \geq 0\\
		\text{0}  & \text{otherwise}
		\end{cases}
		\]

		\end{align*}
